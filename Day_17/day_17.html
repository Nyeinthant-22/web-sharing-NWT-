Day 17: List (Array) ထဲက အချက်အလက်တွေကို ပြောင်းလဲခြင်း

မင်္ဂလာပါ! 👋 မနေ့က List (Array) ထဲက အချက်အလက်တွေကို Index သုံးပြီး ဘယ်လို ရယူရမလဲဆိုတာ လေ့လာခဲ့ကြတယ်နော်။ ဒီနေ့မှာတော့ အဲ့ဒီလို ရယူထားတဲ့ အချက်အလက်တွေကို ပြန်ပြောင်းနည်းကို လေ့လာသွားကြမယ် 😉။

List (Array) ထဲက တန်ဖိုးတွေကို ပြောင်းလဲခြင်း

List (Array) ထဲမှာရှိတဲ့ တန်ဖိုးတွေကို ပြောင်းလဲဖို့အတွက်၊ ပြောင်းချင်တဲ့ အချက်အလက်ရဲ့ Index ကို သုံးပြီး တန်ဖိုးအသစ်တစ်ခု ထည့်ပေးရပါတယ်။

ဥပမာ:

let myNumbers = [10, 20, 30, 40, 50];

console.log(myNumbers); // အစက [10, 20, 30, 40, 50] ဆိုပြီး ပြမယ်

myNumbers[1] = 25; // Index 1 မှာရှိတဲ့ 20 ကို 25 ပြောင်းလိုက်တာ

console.log(myNumbers); // [10, 25, 30, 40, 50] ဆိုပြီး ပြောင်းသွားတာကို တွေ့ရမယ်

myNumbers[3] = myNumbers[3] + 10; // Index 3 မှာရှိတဲ့ 40 ကို 10 ပေါင်းပြီး ပြန်ထည့်လိုက်တာ
console.log(myNumbers);  // [10, 25, 30, 50, 50] ဆိုပြီး ပြောင်းသွားတာကို တွေ့ရမယ်

ရှင်းလင်းချက်

  • myNumbers[1] = 25; ဆိုတာက myNumbers Array ရဲ့ Index 1 နေရာမှာရှိတဲ့ တန်ဖိုးကို 25 နဲ့ အစားထိုးတာပါ။
  • myNumbers[3] = myNumbers[3] + 10; ဆိုတာက Index 3 မှာရှိတဲ့ တန်ဖိုးကို သူ့ထဲကတန်ဖိုး + 10 ပြန်ထည့်တာပါ။

နောက်ထပ် ဥပမာ

let myFruits = ["apple", "banana", "orange"];

myFruits[2] = "grape"; // "orange" ကို "grape" နဲ့ ပြောင်းလိုက်တာ
console.log(myFruits); // ["apple", "banana", "grape"] ဆိုပြီး ပြမယ်

myFruits[0] = myFruits[0].toUpperCase(); // "apple" ကို အကြီးစာလုံး "APPLE" ပြောင်းလိုက်တာ
console.log(myFruits); // ["APPLE", "banana", "grape"] ဆိုပြီး ပြမယ်

ရှင်းလင်းချက်

JavaScript မှာ String တွေမှာလည်း လုပ်ဆောင်ချက်တွေ ရှိပါတယ်။ .toUpperCase() က String တစ်ခုကို အကြီးစာလုံး အဖြစ် ပြောင်းပေးပါတယ်။

ဒီနေ့မှာတော့ List (Array) ထဲက အချက်အလက်တွေကို Index ကို သုံးပြီး ဘယ်လို ပြောင်းလဲရမလဲဆိုတာကို လေ့လာခဲ့ကြပါတယ်။ Array ထဲက တန်ဖိုးတွေကို ပြောင်းလဲခြင်းဟာ Data တွေကို ကိုင်တွယ်တဲ့နေရာမှာ အရေးကြီးတဲ့ အခန်းကဏ္ဍတစ်ခုပါပဲ။ နောက်ရက်တွေမှာတော့ Array တွေနဲ့ ဘာတွေ ဆက်လုပ်လို့ရလဲဆိုတာကို ဆက်ပြီး လေ့လာသွားကြမယ်နော်။ 😊
